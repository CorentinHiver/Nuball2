637 uchar Clovers::label_to_cristal(Label const & l)
638 {
639  auto const cristal = static_cast<uchar>(l-23);
640  auto const clover_cristal = static_cast<uchar>(cristal%6);
641 
642  // BGO :
643  if (clover_cristal<2) return static_cast<uchar>(clover_cristal+2*(cristal/6));
644 
645  // Ge :
646  else
647  {
648  auto const clover = cristal/6; // Clover number
649  auto const Ge_cristal = clover_cristal-2; // Cristal number inside the clover [0,3]
650  auto cristal_index = Ge_cristal+4*clover;
651 
652  // For normally rotated clovers, the following disposition is taken :
653  // With sub_index beeing ranging from 0 to 3 for Ge crystal,
654  // With beam from right to left, and the gamma rays going out of the sheet :
655  // The color green is top right, black top left, red bottom right and blue bottom left for normally rotated clovers
656  // The sub_index 0 is top right, 1 top left, 2 bottom right, 3 bottom_left
657  // That is, the angle phi increases with the sub_index
658  // the even labels are on the right side
659  // the odd labels are on the left side
660 
661  switch (clover)
662  { // Rotation is trigowise direction
663 
664  // pi/2 rotation : R2A6
665  case 17:
666  switch(Ge_cristal)
667  {
668  case 0: return static_cast<uchar>(cristal_index+0); // Red
669  case 1: return static_cast<uchar>(cristal_index+0); // Green
670  case 2: return static_cast<uchar>(cristal_index+1); // Black
671  case 3: return static_cast<uchar>(cristal_index-1); // Blue
672  default : return -1;
673  }
674
