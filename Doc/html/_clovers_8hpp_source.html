182Bool_t Clovers::Fill(Event const & event, int const & hit_i)
183{
184 auto const & label = event.labels[hit_i];
185 if (isClover[label])
186 {
187 auto const & index_clover = labels[label];
188
189 auto const & nrj = event.nrjs [hit_i];
190 auto const & time = event.time2s[hit_i];
191
192 auto & clover = m_Clovers[index_clover];
193
194 // Fill the vector containing the list of all the clovers that fired :
195 Hits.push_back_unique(index_clover);
196
197 if (isGe[label])
198 {
199 CrystalMult++;
200 // Ge crystal index :
201 auto const & index_cristal = cristaux_index[label];
202 // Fill the vector containing the list of all the Ge crystals that fired :
203 cristaux.push_back(index_cristal);
204 // Filling the germanium crystals informations :
205 cristaux_nrj[index_cristal] = nrj;
206 cristaux_time[index_cristal] = time;
207 // Counts the number of Ge crystals that fired in the clover :
208 clover.nb ++;
209 // Finds the Ge cristal that received the most energy and use its time for the index_clover :
210 if (nrj > cristaux_nrj[clover.maxE_Ge_cristal])
211 {
212 clover.maxE_Ge_cristal = index_cristal;
213 clover.time = time;
214 }
215 // Fill the vector containing the list of Ge clovers that fired :
216 Ge.push_back_unique(index_clover);
217 // Do a raw add-back : sum the energy of all the crystals in a clover
218 clover.nrj += event.nrjs[hit_i];
219
220 // Detailed analysis :
221 if (nrj>507 && nrj<516) has511 = true;
222 }
223
224 else
225 {
226 // Cristals managing :
227 CrystalMult_BGO++;
228 // BGO crystal index :
229 auto const & index_cristal = cristaux_index_BGO[label];
230 // Fill the vector containing the list of all the BGO crystals that fired :
231 cristaux_BGO.push_back(index_cristal);
232 // Counts the number of BGO crystals that fired in the clover :
233 clover.nb_BGO ++;
234
235 // Clovers managing :
236 // Fill the vector containing the list of BGO clovers that fired :
237 Bgo.push_back_unique(index_clover);
238 // Fill the cell containing the total energy deposit in the module's BGOs
239 clover.nrj_BGO += event.nrjs[hit_i];
240 }
241 return true;
242 }
243 else return false;
244}

268uchar Clovers::label_to_cristal(Label const & l)
269{
270 auto const cristal = l-23;
271 auto const clover_cristal = cristal%6; // 0 and 1 are BGOs, 2 to 5 are
272 if (clover_cristal<2) return clover_cristal+cristal/6;
273 else
274 {
275 auto const clover = cristal/6;
276 switch (clover) // clover number, to take into account the rotated ones :
277 { // Rotation is trigowise direction
278 // pi rotation
279 case 17:
280 switch(clover_cristal-2)
281 {
282 case 0: return clover; // Red
283 case 1: return clover+1; // Green
284 case 2: return clover+3; // Black
285 case 3: return clover+2; // Blue
286 default : return -1;
287 }
288
289 // -pi rotation : R3A3, R3A4
290 case 2: case 3:
291 switch(clover_cristal-2)
292 {
293 case 0: return clover+3; // Red
294 case 1: return clover+2; // Green
295 case 2: return clover; // Black
296 case 3: return clover+1; // Blue
297 default : return -1;
298 }
299
300 // 2pi rotation
301 case 4:
302 switch(clover_cristal-2)
303 {
304 case 0: return clover+1; // Red
305 case 1: return clover+3; // Green
306 case 2: return clover+2; // Black
307 case 3: return clover+0; // Blue
308 default : return -1;
309 }
310
311 default: // All the correctly turned clovers:
312 switch(clover_cristal-2)
313 {
314 case 0: return clover+2; // Red
315 case 1: return clover; // Green
316 case 2: return clover+1; // Black
317 case 3: return clover+3; // Blue
318 default : return -1;
319 }
320 }
321 }
