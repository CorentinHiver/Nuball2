223bool Sorted_Event::sortGeClover(Event const & event, int const & i)
224{
225 auto const & label = event.labels[i];
226 auto const & nrj = event.nrjs[i];
227 auto const & clover_label = labelToClover_fast[label];
228
229 if (nrj<5) return false;
230
231 push_back_unique(clover_hits, clover_label);
232
233 nrj_clover[clover_label] += nrj;
234 Ge[clover_label]++;
235
236 // To get the crystal that got the maximum energy in a clover :
237 if(nrj > maxE[clover_label])
238 {
239 if (event.readTime()) time_clover[clover_label] = times[i];
240 maxE[clover_label] = nrj;
241 maxE_hit[clover_label] = i;
242 }
243 return true;
244}

246void Sorted_Event::sortEvent(Event const & event)
247{
248 this->reset();
249 RawMult = event.size();
250
251 uchar clover_label = 0;
252 for (uchar i = 0; i<event.size(); i++)
253 {
254 if (event.readtime()) times[i] = (m_rf) ? m_rf->pulse_ToF(event.times[i], 50000ll) : (event.times[i]-event.times[0])/_ns;
255 if (event.readTime()) times[i] = event.time2s[i]; // Overwrites the absolute time if relative Time is read in the data (parameter 'T' in connect() Event's method)
256 if (times[i]>m_max_time) continue;
257 if (isGe[event.labels[i]])
258 {
259 if (!sortGeClover(event, i)) continue;
260 RawGeMult++;
261 }
262 else if (isBGO[event.labels[i]])
263 {
264 clover_label = labelToClover_fast[event.labels[i]];
265 if (m_rf) time_clover[clover_label] = m_rf->pulse_ToF(event.times[i], 50000ll)/_ns;
266 else time_clover[clover_label] = (event.times[i]-event.times[0])/_ns;
267 BGO[clover_label]++;
268 }
269 else if (isLaBr3[event.labels[i]])
270 {
271 LaBr3_Hit labr3_hit;
272 labr3_hit.nrjcal = event.nrjs[i];
273 labr3_hit.label = event.labels[i]-199;
274 if (m_rf) labr3_hit.time = m_rf->pulse_ToF(event.times[i], 50000ll)/_ns;
275 else labr3_hit.time = (event.times[i]-event.times[0])/_ns;
276 LaBr3_event.push_back(labr3_hit);
277 ModulesMult++;
278 LaBr3Mult++;
279 }
280 else if (isParis[event.labels[i]])
281 {
282 paris_hits.push_back(i);
283 ModulesMult++;
284 LaBr3Mult++;
285 ParisMult++;
286 }
287 else if (isDSSD[event.labels[i]])
288 {
289 DSSD_hits.push_back(i);
290 auto const & nrj = event.nrjs[i];
291 auto const & label = event.labels[i];
292 #ifdef N_SI_129
293 auto const & time = times[i];
294 if (time>-5 && time<15 && nrj>8500 && nrj<11000) DSSDFirstBlob = true;
295 else if (time> 3 && time<18 && nrj>5500 && nrj<7500) DSSDSecondBlob = true;
296 else if (time>10 && time<20 && nrj>3800 && nrj<5500) DSSDThirdBlob = true;
297 else if (time>25 && time<70 && nrj>2000 && nrj<3500) DSSDTail = true;
298 #endif //N_SI_129
299 if (m_TW_DSSD)
300 {
301 times[i] -= m_Timewalks_DSSD.get(label-800, nrj);
302 DSSD_is_Prompt.push_back((times[i]>-10 && times[i]<10));
303 if (times[i]<m_min_time_dssd || times[i]>m_max_time_dssd || nrj>m_max_E_dssd) dssd_veto = true;
304 }
305 else DSSD_is_Prompt.push_back((times[i]>0 && times[i]<50));
306
307
308 bool const & isRing = isDSSD_Ring[event.labels[i]];
309 DSSD_is_Ring.push_back(isRing);
310 if (isRing)
311 {
312 DSSD_Rings.push_back(i);
313 DSSDRingMult++;
314 }
315 else
316 {
317 DSSD_Sectors.push_back(i);
318 DSSDSectorMult++;
319 }
320 DSSDMult++;
321 }
322 }
323
324 // Add-Back and Compton Cleaning of Clovers :
325 for (size_t clover = 0; clover<24l; clover++)
326 {
327 if (Ge[clover])
328 {
329 if (time_clover[clover_label]>-10 && time_clover[clover_label]<70) prompt_Ge[clover_label] = true;
330 else if (time_clover[clover_label]>70) delayed_Ge[clover_label] = true;
331 CloverMult++;
332 ModulesMult++;
333 CloverGeMult++;
334 if (BGO[clover]) ComptonVetoMult++;
335 else
336 {
337 CleanGeMult++;
338 // Energy gates :
339 if (m_gateON && m_Ge_gate == 0) for (size_t gate = 0; gate<m_Ge_gates.size(); gate++)
340 {
341 if (nrj_clover[clover]>m_Ge_gates[gate].first && nrj_clover[clover]<m_Ge_gates[gate].second)
342 {
343 m_Ge_gate = gate+1;
344 m_CloverTrig = clover;
345 break;
346 }
347 }
348 }
349 }
350 else if (BGO[clover])
351 {
352 CloverMult++;
353 ModulesMult++;
354 BGOMult++;
355 }
356 }
357
358 //Analyse DSSD :
359 for (auto const & prompt : DSSD_is_Prompt)
360 {
361 if (prompt) DSSDPrompt = true;
362 else {DSSDPrompt = false; break;}
363 }
364}
