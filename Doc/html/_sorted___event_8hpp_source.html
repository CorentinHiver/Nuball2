254 void Sorted_Event::sortEvent(Event const & event)
255 {
256  this->reset();
257  RawMult = event.size();
258 
259  uchar clover_label = 0;
260  for (uchar i = 0; i<event.size(); i++)
261  {
262  if (event.read.t) time2s[i] = (m_rf) ? m_rf->pulse_ToF(event.time2s[i], 50000ll) : (event.time2s[i]-event.time2s[0])/_ns;
263  if (event.read.T) time2s[i] = event.time2s[i]; // Overwrites the absolute time if relative Time is read in the data (parameter 'T' in connect() Event's method)
264  if (time2s[i]>m_max_time) continue;
265  if (isGe[event.labels[i]])
266  {
267  if (!sortGeClover(event, i)) continue;
268  RawGeMult++;
269  }
270  else if (isBGO[event.labels[i]])
271  {
272  clover_label = labelToClover[event.labels[i]];
273  if (m_rf) time_clover[clover_label] = m_rf->pulse_ToF(event.time2s[i], 50000ll)/_ns;
274  else time_clover[clover_label] = (event.time2s[i]-event.time2s[0])/_ns;
275  BGO[clover_label]++;
276  }
277  else if (isLaBr3[event.labels[i]])
278  {
279  LaBr3_Hit labr3_hit;
280  labr3_hit.nrjcal = event.nrjs[i];
281  labr3_hit.label = event.labels[i]-199;
282  if (m_rf) labr3_hit.time = m_rf->pulse_ToF(event.time2s[i], 50000ll)/_ns;
283  else labr3_hit.time = (event.time2s[i]-event.time2s[0])/_ns;
284  LaBr3_event.push_back(labr3_hit);
285  ModulesMult++;
286  LaBr3Mult++;
287  }
288  else if (isParis[event.labels[i]])
289  {
290  paris_hits.push_back(i);
291  ModulesMult++;
292  LaBr3Mult++;
293  ParisMult++;
294  }
295  else if (isDSSD[event.labels[i]])
296  {
297  DSSD_hits.push_back(i);
298  auto const & nrj = event.nrjs[i];
299  auto const & label = event.labels[i];
300  #ifdef N_SI_129
301  auto const & time = time2s[i];
302  if (time>-5 && time<15 && nrj>8500 && nrj<11000) DSSDFirstBlob = true;
303  else if (time> 3 && time<18 && nrj>5500 && nrj<7500) DSSDSecondBlob = true;
304  else if (time>10 && time<20 && nrj>3800 && nrj<5500) DSSDThirdBlob = true;
305  else if (time>25 && time<70 && nrj>2000 && nrj<3500) DSSDTail = true;
306  #endif //N_SI_129
307  if (m_TW_DSSD)
308  {
309  time2s[i] -= m_Timewalks_DSSD.get(label-800, nrj);
310  DSSD_is_Prompt.push_back((time2s[i]>-10 && time2s[i]<10));
311  if (time2s[i]<m_min_time_dssd || time2s[i]>m_max_time_dssd || nrj>m_max_E_dssd) dssd_veto = true;
312  }
313  else DSSD_is_Prompt.push_back((time2s[i]>0 && time2s[i]<50));
314 
315 
316  bool const & is_ring = isRing[event.labels[i]];
317  DSSD_is_Ring.push_back(is_ring);
318  if (is_ring)
319  {
320  DSSD_Rings.push_back(i);
321  DSSDRingMult++;
322  }
323  else
324  {
325  DSSD_Sectors.push_back(i);
326  DSSDSectorMult++;
327  }
328  DSSDMult++;
329  }
330  }
331 
332  // Add-Back and Compton Cleaning of Clovers :
333  for (size_t clover = 0; clover<24l; clover++)
334  {
335  if (Ge[clover])
336  {
337  if (time_clover[clover_label]>-10 && time_clover[clover_label]<70) prompt_Ge[clover_label] = true;
338  else if (time_clover[clover_label]>70) delayed_Ge[clover_label] = true;
339  CloverMult++;
340  ModulesMult++;
341  CloverGeMult++;
342  if (BGO[clover]) ComptonVetoMult++;
343  else
344  {
345  CleanGeMult++;
346  // Energy gates :
347  if (m_gateON && m_Ge_gate == 0) for (size_t gate = 0; gate<m_Ge_gates.size(); gate++)
348  {
349  if (nrj_clover[clover]>m_Ge_gates[gate].first && nrj_clover[clover]<m_Ge_gates[gate].second)
350  {
351  m_Ge_gate = gate+1;
352  m_CloverTrig = clover;
353  break;
354  }
355  }
356  }
357  }
358  else if (BGO[clover])
359  {
360  CloverMult++;
361  ModulesMult++;
362  BGOMult++;
363  }
364  }
365 
366  //Analyse DSSD :
367  for (auto const & prompt : DSSD_is_Prompt)
368  {
369  if (prompt) DSSDPrompt = true;
370  else {DSSDPrompt = false; break;}
371  }
372 }
