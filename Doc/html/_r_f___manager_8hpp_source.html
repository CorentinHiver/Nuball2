14 Long64_t pulse_ToF(ULong64_t const & time, Long64_t const & offset = 0) const
15 {
16 #ifdef USE_RF
17 // Shifts the time in order to be able to get hits before the hit :
18 ULong64_t const shifted_time = time+offset;
19 if (shifted_time>last_hit)
20 {// Normal case : the RF reference timestamp is lower than the shifted timestamps (should also be the case of unshifted timestamp)
21 // dT = (shifted_time-last_hit) corresponds to the time separating the current hit to the reference RF
22 // Therefore, N = dT/period corresponds to the number of periods separating the two hits
23 // In other words, it is the number of pulses between the reference RF and the current hit
24 // Then, period*N is the timestamp of the pulse relative to the current hit
25 // (Remember we are doing integer arithmetic, period*dT/period != dT)
26 // And dT%period is the rest of the integer division, hence the time between the hit and its relative pulse
27 // Finally, one need to substract the applied offset in order to get the correct result :
28 return ( static_cast<Long64_t> ((time+offset-last_hit)%period)-offset );
29 }
30 else
31 {// When the data is not correctly ordered :
32 // In order to get a correct answer, one need to get a positive difference, so to invert the difference : now last_hit-shifted_time>0
33 // But the result is inverted and we obtain really period-timestamp. We get the correct result by doing :
34 // relative_timestamp = period - (period-timestamp)
35 return ( static_cast<Long64_t> (period-(last_hit-time-offset)%period)-offset );
36 }
37 #else //NO USE_RF
38 print("NO RF IS USED !! Please set RF_USE [period_value]");
39 #endif //USE_RF
40 }
