93inline float Calibration::calibrate(float const & nrj, Label const & label)
94{
95 auto nrj_r = nrj+gRandom->Uniform(0,1);
96 switch(m_order[label])
97 {
98 case 0:
99 return nrj_r;
100
101 case 1:
102 return m_intercept[label]
103 + m_slope[label] * nrj_r;
104
105 case 2:
106 return m_intercept[label]
107 + m_slope[label] * nrj_r
108 + m_binom[label] * nrj_r * nrj_r;
109
110 case 3:
111 return m_intercept[label]
112 + m_slope [label] * nrj_r
113 + m_binom [label] * nrj_r * nrj_r
114 + m_trinom[label] * nrj_r * nrj_r * nrj_r;
115
116 default:
117 return nrj;
118 }
119}

121inline void Calibration::calibrate(Hit & hit)
122{
123 auto const & label = hit.label;
124 if (label > m_max_labels) return;
125
126#if defined (LICORNE)
127 if (is_EDEN(label))
128 {

130 hit.nrjcal = static_cast<float>(hit.nrj2)/hit.nrj;
131 }
132#elif defined (PARIS)
133 if (isParis[label]) hit.nrj2 = calibrate(hit.nrj2, label);

135

137 else hit.nrjcal = calibrate(hit.nrj, label);

166bool Calibration::load(std::string const & calibFileName, int const & label_max)
167{
168 m_max_labels = label_max;
169 std::ifstream inputfile;
170 inputfile.open(calibFileName);
171 if (!inputfile.good()) return false;
172 std::string line = "";
173 UShort_t label = 0;
174 m_order .resize(label_max);
175 m_intercept.resize(label_max);
176 m_slope .resize(label_max, 1.f); //Fill with 1
177 m_binom .resize(label_max);
178 m_trinom .resize(label_max);
179 float intercept = 0.f, slope = 1.f, binom = 0.f, trinom = 0.f;
180 while (getline(inputfile, line))
181 {
182 m_nb_labels++;
183 std::istringstream iss(line);
184 iss >> label >> intercept >> slope >> binom >> trinom;
185 this -> set(label, intercept, slope, binom, trinom);
186 intercept = 0.f; slope = 1.f; binom = 0.f; trinom = 0.f;
187 }
188 m_isFilled = true;
189 print("Timeshifts extracted from", calibFileName);
190 return true;
191}
