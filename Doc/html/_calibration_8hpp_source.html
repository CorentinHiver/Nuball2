178 inline float Calibration::apply(float const & nrj, Label const & label) const noexcept
179 {
180  // Applies the calibration coefficients to the giver value
181  switch(m_order[label])
182  {
183  case -1: return nrj; // No calibration
184  case 0: return m_slope[label]*nrj; // Linear calibration (no offset)
185  case 1: return m_intercept[label] + m_slope[label]*nrj; // Affine calibration
186  case 2: return m_intercept[label] + m_slope[label]*nrj + m_binom[label]*nrj*nrj; // Quadratic calibration
187  case 3: return m_intercept[label] + m_slope[label]*nrj + m_binom[label]*nrj*nrj + m_trinom[label]*nrj*nrj*nrj; // Third order calibration
188  default: return -1; // This should normally never happen
189  }
190 }
