148Bool_t EventBuilder::build(Hit const & hit)
149{//return true when an event is ready to be processed
150 if (m_status == 2 || mult>254) this->flush(); //because nb_evts is on uchar so < 255
151 if (!coincON)
152 {// No coincidence going on. If there is, go ...
153 if (coincidence(hit))
154 {// Bingo, coincidence detected ! Next call, go ...
155 m_buffer -> push_back( m_last_hit );
156 m_buffer -> push_back( hit );
157 mult = 2;
158 coincON = true;
159 m_status = 1;
160 }
161 else
162 {// No coincidence detected...
163 m_single_hit = m_last_hit;
164 m_last_hit = hit; //store the hit for next loop
165 mult = 1;
166 n_evt++;
167 m_status = 0;
168 }
169 }
170 else
171 {// ... there ! Coincidence already detected, check if it ends
172 if (coincidence(hit))
173 {// Hit in coincidence with the previous hits
174 mult++;
175 m_buffer -> push_back(hit);
176 }
177 else
178 {// Hit out of coincidence. Next call, go back to first condition.
179 n_evt++;
180 coincON = false;
181 m_last_hit = hit;
182 m_status = 2;
183 return true;
184 }
185 }
186 return false;
187}
