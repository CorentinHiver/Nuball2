679  {
680  // 1.a Calculate the chi2 for each vertice of the simplex :
681  std::vector<double> chi2s;

698 
699  // 2.a Compute the reflexion point : centroid + alpha * (centroid-worst_vertice)

708  Vertice expansion(centroid + nmParam.beta*(reflection-centroid));
709  // 3.b Compute its chi2

711  print("chi2_expansion", chi2_expansion);

713  else simplex[ordered_indexes.back()] = reflection;
714  }

716  {
717  simplex[ordered_indexes.back()] = reflection;
718  }
719  else if (chi2_reflection < ordered_chi2s.back())
720  {
721  // 4.a Compute the outside contraction point : centroid + beta * (reflection-centroid)
722  Vertice outsideContraction(centroid + nmParam.gamma*(reflection-centroid));

727  }

729  {
730  // 5.a Compute the inside contraction point : centroid + beta * (reflection-centroid)
731  Vertice insideContraction(centroid - nmParam.gamma*(reflection-centroid));
732  // 5.b Compute its chi2
733  auto chi2_inside_contraction = chi2(ref_spectra, spectra, insideContraction);
734  print("chi2_inside_contraction", chi2_inside_contraction);
735  if (chi2_inside_contraction < chi2_reflection) simplex[ordered_indexes.back()] = insideContraction;
736  }

739  Simplex tempSimplex(simplex);
740  tempSimplex[0] = simplex[ordered_indexes[0]]; // this is the best point
741  for (size_t point_i = 1; point_i<dim+1; point_i++)
742  {
743  tempSimplex[point_i] = tempSimplex[0] + nmParam.delta * (simplex[ordered_indexes[point_i]] - tempSimplex[0]);
744  }

748  print(simplex);
