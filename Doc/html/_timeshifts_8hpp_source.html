21/*
22 * Author : Corentin Hiver
23 * Module part of the NearLine software
24 * The timestamps are in picoseconds, so are the timeshifts
25 * Goal 1 : calculate the time shifts between detectors
26 * How :
27 * * Use one detector as reference, preferencially with good time resolution
28 * * Create coincidences using the CoincBuilder event builder
29 * * Calculate the difference in timestamp as follow : ref_time - detector_time
30 * * Finally, output the (.dT) file containing the time shifts of all the detectors
31 * * Also produces a .root file containing the raw and corrected time spectra.
32 * * This module includes multithreading management using MTObject::parallelise_function(function, parameters...)
33 * Goal 2 : read a timeshift file (.dT) previously calculated by this module
34 * Goal 3 : apply the timeshifts by calling the operator[] : time_correct = time_shifted + timeshifts[label];
35 * Example at the end
36*/

225bool Timeshifts::load(std::string const & filename)
226{
227 std::ifstream inputfile(filename, std::ifstream::in);
228 if (!inputfile.is_open())
229 {
230 print("Could not open the time shifts file - '", filename, "'");
231 return (m_ok = false);
232 }
233 else if (file_is_empty(inputfile))
234 {
235 print("Time shift file - '", filename, "' is empty !");
236 return (m_ok = false);
237 }
238 // ----------------------------------------------------- //
239 UShort_t size = 0; std::string line = ""; int label = 0;
240 while (getline(inputfile, line))
241 { // First extract the maximum label
242 std::istringstream iss(line);
243 iss >> label;
244 if (size<label) size = label;
245 }
246 // ----------------------------------------------------- //
247 // Second reading : fill the vector
248 m_timeshifts.resize(size+1);
249 inputfile.clear(); inputfile.seekg(0, inputfile.beg);
250 int deltaT = 0;
251 while (getline(inputfile, line))
252 { // Then fill the array
253 std::istringstream iss(line);
254 iss >> label >> deltaT;
255 m_timeshifts[label] = deltaT;
256 }
257 inputfile.close();
258 print("Timeshifts extracted from", filename);
259 return (m_ok = true);
260}

337bool Timeshifts::calculate(std::string const & folder, int const & nb_files)
338{
339 Timer timer;
340 if (!InitParameters())
341 {
342 print("Someting went wrong in the timeshifts initialization...");
343 return false;
344 }
345
346 m_files.addFolder(folder, nb_files);
347
348 if (MTObject::ON)
349 {// If multithreading, treat each data file of the folder in parallel
350 print("Calculating timeshifts with", MTObject::getThreadsNb(),"threads");
351 // The FileManager object isn't thread safe.
352 // That is why one has to encapsulate the files list inside a MTList (Multi-Threaded List) :
353 MTList<std::string> files_MT(m_files.getListFiles());
354 MTObject::parallelise_function(treatFilesMT, *this, files_MT);
355 }
356 else
357 {// If no multithreading, treat each data file sequentially
358 print("Calculating timeshifts without multithreading");
359 std::string filename;
360 while (m_files.nextFileName(filename))
361 {
362 treatFile(filename);
363 }
364 }
365
366 // Once the histograms have been filled, fit the peaks
367 this -> analyse();
368
369 // Write down the timeshitfs file in a .dT file, and the root histograms in a .root file
370 write();
371
372 print("Timeshifts calculated in", timer(), timer.unit());
373
374 return true;
375}

384 while (files_MT.getNext(filename))
385 {
386 ts.treatFile(filename);

402void Timeshifts::treatFile(std::string const & filename)
403{
404 if (extension(filename) == ".root") {treatRootFile(filename); return;}
405 Hit hit;
406 FasterReader reader(&hit, filename);
407 Event event;
408 CoincBuilder2 coincBuilder(&event, m_timewindow);
409
410
411 uint counter = 0;
412
413 bool maxHitsToRead = (m_max_hits>0);
414
415#ifdef USE_RF
416 RF_Manager rf;
417// get_first_RF_of_file()
418#endif //USE_RF
419
420 while(reader.Read() && ((maxHitsToRead) ? (counter<m_max_hits) : true) )
421 {
422 if ( (counter%static_cast<uint>(1.E+6)) == 0 ) print(counter);
423 counter++;
424
425 #ifdef USE_RF
426 if(isRF[hit.label]) {rf.last_hit = hit.time; rf.period = hit.nrj;}
427 else if (hit.label == m_timeRef_label)
428 {
429 hit.nrjcal = hit.nrj;
430 m_histo_ref_VS_RF.Fill(rf.pulse_ToF(hit.time));
431 }
432 else
433 {
434 m_histograms_VS_RF[hit.label].Fill(rf.pulse_ToF(hit.time, USE_RF));
435 }
436 #endif //USE_RF
437
438 if (coincBuilder.build(hit))
439 {
440 auto mult = event.size();
441 if (mult >= m_max_mult && mult <= m_min_mult) continue;
442 for (size_t i = 0; i < mult; i++)
443 {
444 if ( event.labels[i] == m_timeRef_label ) // if the ref detector triggered
445 {
446 auto const & refPos = i;
447 Shift_t deltaT = 0;
448 m_EnergyRef.Fill(event.nrjs[refPos]);
449 for (uchar j = 0; j<mult; j++)
450 {
451 if (event.labels[j] == m_timeRef_label) continue; // To reject the time spectra of the time reference
452 deltaT = event.times[refPos]-event.times[j];
453 m_histograms[event.labels[j]].Fill(deltaT); //stored in ps
454}}}}}}

472bool getMeanPeak(TH1F* spectra, double & mean)
473{
474 // Declaration :
475 std::unique_ptr<TF1> gaus_pol0;
476 std::unique_ptr<TF1> fittedPic;
477 float pospic, amppic, dump_sigma;
478 float cte, Mean, sigma;
479
480 // Histogram characteristics :
481 auto const bins = spectra -> GetXaxis() -> GetNbins();
482 auto const xmax = spectra -> GetXaxis() -> GetXmax();
483 auto const xmin = spectra -> GetXaxis() -> GetXmin();
484
485 auto const xPerBin = (xmax-xmin)/bins;
486 auto const bin0 = getBin0(spectra);
487
488 // Extract dump parameters :
489 amppic = spectra -> GetMaximum();
490 pospic = static_cast<double>( (spectra->GetMaximumBin() - bin0)*xPerBin );
491 dump_sigma = static_cast<double>( (spectra->FindLastBinAbove(amppic/2) - spectra->FindFirstBinAbove(amppic/2)) * xPerBin/2 );
492
493 // Fits the peak :
494 gaus_pol0.reset(new TF1("gaus+pol0","gaus(0)+pol0(3)",pospic-20*dump_sigma,pospic+20*dump_sigma));
495 gaus_pol0 -> SetParameters(amppic, pospic, dump_sigma, 1);
496 gaus_pol0 -> SetRange(pospic-dump_sigma*20,pospic+dump_sigma*20);
497 spectra -> Fit(gaus_pol0.get(),"R+q");
498
499 // Extracts the fitted parameters :
500 fittedPic.reset (spectra -> GetFunction("gaus+pol0"));
501 if (!fittedPic) return false; // Eliminate non existing fits, when not enough statistics fit doesn't converge
502 cte = fittedPic -> GetParameter(0);
503 mean = Mean = fittedPic -> GetParameter(1);
504 sigma = fittedPic -> GetParameter(2);
505
506 if (false) print("cte", cte, "Mean", Mean, "sigma", sigma);
507
508 return true;
509}

511void Timeshifts::analyse()
512{
513 print("Get the timeshifts from the time spectra");
514 std::vector<float> array_labels(m_detList.size());
515 for (size_t i = 0; i<array_labels.size(); i++) array_labels[i] = i;
516 std::ofstream outDeltaTfile(m_outPath+m_outData, std::ios::out);
517
518#ifdef USE_RF
519 // Calculate the RF timeshift first :
520 m_histo_ref_VS_RF.Merge();
521 m_timeshifts[RF_Manager::label] = static_cast<Shift_t>(( m_histo_ref_VS_RF->GetMaximumBin() - (m_histo_ref_VS_RF -> GetNbinsX()/2) ) * m_rebin["RF"]);
522#endif //USE_RF
523
524 for (size_t label = 0; label<m_detList.size(); label++)
525 {
526 auto const & type = Detectors::type(label);
527 if (type == "null") continue;
528 m_histograms[label].Merge();
529
530 #ifdef USE_RF
531 m_histograms_VS_RF[label].Merge();
532 if (label == RF_Manager::label)
533 {
534 if (m_verbose) print("RF :", m_timeshifts[RF_Manager::label], "with", m_histo_ref_VS_RF->GetMaximum(), "bins in peak");
535 m_histo_RF_corrected.reset(shiftTimeSpectra(m_histo_ref_VS_RF, 251));
536 outDeltaTfile << label << "\t" << m_timeshifts[RF_Manager::label] << std::endl;
537 continue;
538 }
539 #endif //USE_RF
540
541 if (!m_histograms[label].exists()) continue; // Eliminate empty histograms
542 if (m_verbose) print(m_detList[label]);
543
544 #ifdef LICORNE
545 if (type == "Eden")
546 {
547 outDeltaTfile << label << "\t" << 0;
548 continue;
549 }
550 #endif //LICORNE
551
552 #ifdef USE_DSSD
553 if (type == "dssd")
554 {
555 #ifdef USE_RF
556 auto amppic = m_histograms_VS_RF[label] -> GetMaximum();
557 float zero = ( m_histograms_VS_RF[label] -> FindFirstBinAbove(amppic/2) - (m_histograms_VS_RF[label] -> GetNbinsX()/2) ) * m_rebin["dssd"] ;
558 outDeltaTfile << label << "\t" << static_cast<float>(m_timeshifts[RF_Manager::label]-zero) << std::endl;
559 if (m_verbose) print("Edge :", zero, "with", static_cast<int>(m_histograms_VS_RF[label] -> GetMaximum()), "counts in peak");
560
561 #else //NO USE_RF
562 auto amppic = m_histograms[label] -> GetMaximum();
563 auto pospic = ( m_histograms[label] -> FindLastBinAbove(amppic*0.8) - (m_histograms[label] -> GetNbinsX()/2) ) * m_rebin["dssd"] ;
564 outDeltaTfile << label << "\t" << pospic << std::endl;
565 if (m_verbose) print( "mean : ", m_histograms[label] -> GetMean(), "with", (int) m_histograms[label] -> GetMaximum(), "counts in peak");
566 #endif //USE_RF
567
568 continue;
569 }
570 #endif
571
572 if (label == m_timeRef_label) m_timeshifts[label] = 0; // Force the reference detector to be at 0;
573 else
574 {
575 double mean = 0;
576 if (getMeanPeak(m_histograms[label], mean)) m_timeshifts[label] = static_cast<Shift_t> (-mean);
577 }
578
579 outDeltaTfile << label << "\t" << m_timeshifts[label] << std::endl;
580
581 }
582 // Now create the corrected time spectra :
583 for (size_t label = 0; label<m_histograms.size(); label++)
584 {
585 auto & histo = m_histograms[label];
586 if (histo)
587 {
588 m_histograms_corrected[label].reset( shiftTimeSpectra(histo, label) );
589 }
590
591 #ifdef USE_RF
592 auto & histo_RF = m_histograms_VS_RF[label];
593 if (histo_RF) m_histograms_corrected_RF[label].reset( shiftTimeSpectra(histo_RF, label) );
594 #endif //USE_RF
595 }
596
597 outDeltaTfile.close();
598
599 std::cout << "Timeshifts data written to " << m_outPath+m_outData << std::endl;
600}

684TH1F* Timeshifts::shiftTimeSpectra(TH1F* histo, Label const & label, std::string const & unit)
685{
686 std::string name = histo->GetName();
687 if (name == "") return (new TH1F());
688
689 auto axis = histo -> GetXaxis();
690
691 auto bins = axis -> GetNbins();
692 auto timeMax = axis -> GetXmax ();
693 auto timeMin = axis -> GetXmin ();
694
695 auto binToTime = (timeMax-timeMin)/bins;
696 auto time_to_ps = 1;
697
698 if (unit != "ps")
699 {
700 if (unit == "ns") time_to_ps*=1000;
701 else if (unit == "ticks") time_to_ps*=128; // Deprecated
702 else print("UNKOWN TIME UNIT", unit);
703 return nullptr;
704 }
705
706 auto corrected_histo = static_cast<TH1F*>(histo -> Clone((name+"_corrected").c_str()));
707
708 for (int bin = 1; bin<bins; bin++)
709 {
