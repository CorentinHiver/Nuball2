21 /*
22  * Author : Corentin Hiver
23  * Module part of the NearLine software
24  * The timestamps are in picoseconds, so are the timeshifts
25  * Goal 1 : calculate the time shifts between detectors
26  * How :
27  * * Use one detector as reference, preferencially with good time resolution
28  * * Create coincidences using the CoincBuilder event builder
29  * * Calculate the difference in timestamp as follow : ref_time - detector_time
30  * * Finally, output the (.dT) file containing the time shifts of all the detectors
31  * * Also produces a .root file containing the raw and corrected time spectra.
32  * * This module includes multithreading management using MTObject::parallelise_function(function, parameters...)
33  * Goal 2 : read a timeshift file (.dT) previously calculated by this module
34  * Goal 3 : apply the timeshifts by calling the operator[] : time_correct = time_shifted + timeshifts[label];
35  * Example at the end
36 */

513 {
514  // Declaration :
515  std::unique_ptr<TF1> gaus_pol0;
516  std::unique_ptr<TF1> fittedPic;
517  float pospic, amppic, dump_sigma;
518  float cte, Mean, sigma;
519 
520  // Histogram characteristics :
521  auto const bins = spectra -> GetXaxis() -> GetNbins();
522  auto const xmax = spectra -> GetXaxis() -> GetXmax();
523  auto const xmin = spectra -> GetXaxis() -> GetXmin();
524 
525  auto const xPerBin = (xmax-xmin)/bins;
526  auto const bin0 = getBin0(spectra);
527 
528  // Extract dump parameters :
529  amppic = spectra -> GetMaximum();
530  pospic = static_cast<double>( (spectra->GetMaximumBin() - bin0)*xPerBin );
531  dump_sigma = static_cast<double>( (spectra->FindLastBinAbove(amppic/2) - spectra->FindFirstBinAbove(amppic/2)) * xPerBin/2 );
532 
533  // Fits the peak :
534  gaus_pol0.reset(new TF1("gaus+pol0","gaus(0)+pol0(3)",pospic-20*dump_sigma,pospic+20*dump_sigma));
535  gaus_pol0 -> SetParameters(amppic, pospic, dump_sigma, 1);
536  gaus_pol0 -> SetRange(pospic-dump_sigma*20,pospic+dump_sigma*20);
537  spectra -> Fit(gaus_pol0.get(),"R+q");
538 
539  // Extracts the fitted parameters :
540  fittedPic.reset (spectra -> GetFunction("gaus+pol0"));
541  if (!fittedPic) return false; // Eliminate non existing fits, when not enough statistics fit doesn't converge

543  mean = Mean = fittedPic -> GetParameter(1);
544  sigma = fittedPic -> GetParameter(2);

745 TH1F* Timeshifts::shiftTimeSpectra(TH1F* histo, Label const & label, std::string const & unit)
746 {
747  std::string name = histo->GetName();
748  if (name == "") return (new TH1F());
749 
750  auto axis = histo -> GetXaxis();
751 
752  auto bins = axis -> GetNbins();
753  auto timeMax = axis -> GetXmax ();
754  auto timeMin = axis -> GetXmin ();
755 
756  auto binToTime = (timeMax-timeMin)/bins;
757  auto time_to_ps = 1;
758 
759  if (unit != "ps")
760  {
761  if (unit == "ns") time_to_ps*=1000;
762  else if (unit == "ticks") time_to_ps*=128; // Deprecated
763  else print("UNKOWN TIME UNIT", unit);
764  return nullptr;
765  }
766 
767  auto corrected_histo = static_cast<TH1F*>(histo -> Clone((name+"_corrected").c_str()));
768 
769  for (int bin = 1; bin<bins; bin++)
770  {
771  auto const time_ps = time_to_ps*(binToTime*bin + timeMin); // time_to_ps * (coeff*bin + offset_time)
772  auto const new_time_ps = time_ps + m_timeshifts[label];
773  auto const new_time = new_time_ps/time_to_ps;
774  int const newbin = ( new_time - timeMin ) / binToTime; // ( (time_ps/time_to_ps) - offset_time) / coeff
775 
776  if( newbin>0 && newbin < bins )
777  {
778  corrected_histo -> SetBinContent(newbin, histo->GetBinContent(bin));
779  }
780  }
781  return corrected_histo;
782 }
