211 void Faster2Root::load(int argc, char** argv)
212 {
213  // Handle the case of not enough parameters (0 or 1) and print the parameters :
214  if (argc<3)
215  {
216  print("Not enough parameters !!!");
217  printParameters();
218  }
219 
220  // Path of the input data :
221  Path dataPath = argv [1];
222 
223  // Path of the output data :
224  Path outPath = argv [2];
225 
226  // Other parameters :
227  for (int i = 3; i<argc; i++)
228  {
229  std::string command = argv[i];
230  if (command == "-c") loadCalibration(argv[++i]);
231  else if (command == "-e") buildEvents(std::stoi(argv[++i]));
232  else if (command == "-f") setNbFiles(std::stoi(argv[++i]));
233  else if (command == "-i") detectors.load(argv[++i]);
234  else if (command == "-m") setNbThreads(std::stoi(argv[++i]));
235  else if (command == "-n") FasterReader::setMaxHits(std::stoi(argv[++i]));
236  else if (command == "-o") overwrite(true);
237  else if (command == "-rf") m_use_RF = true; // TBD
238  else if (command == "-t") loadTimeshifts(argv[++i]);
239  else if (command == "--throw-singles") m_throw_single = true;
240  else if (command == "--trigger") loadTriggerFile(argv[++i]);
241  else {throw_error("Unkown command" + command);}
242  }
243 
244  // Checking consistency of parameters :
245  if (m_timeshifts && !m_eventbuilding)
246  {
247  print("This combination of time alignement without event building is not handled");
248  throw_error(error_message["DEV"]);
249  }
250  if (m_use_trigger && !m_eventbuilding)
251  {
252  throw_error("Can't use a trigger without event building ! Use parameter -e [time_window_ns] or method Faster2Root::buildEvents(time_window_ns).");
253  }
254 
255  // Perform initialisations :
256  if (m_nb_threads > 1) MTObject::Initialise(m_nb_threads);
257  std::cout << std::setprecision(4);
258 
259  // Run the conversion :
260  this -> convert(dataPath.string(), outPath.string());
261 
262  // Print some additionnal informations at the end of the conversion :
263  print(m_total_hits/m_total_timer.TimeSec()*1.E-6, "Mhits/s");
