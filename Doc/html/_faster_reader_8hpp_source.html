314 bool FasterReader::ReadGroup()
315 {
316  m_write = false;
317  while(!m_write)
318  {
319  if (m_inGroup)
320  { // We currently are reading a group
321  // if (m_group_read_cursor > m_group_write_cursor) // Might be losing the last hit with "==" condition ...
322  if (m_group_read_cursor == m_group_write_cursor)
323  { // We are getting out of the group, time to read the next data or group of data
324  m_inGroup = false;
325  // Reset m_hit_group_buffer cursors
326  m_group_read_cursor = 0;
327  m_group_write_cursor = 0;
328  m_data = faster_file_reader_next(m_reader);
329  if (!m_data) return false; // This tags the end of the data
330  ReadData(m_data);
331  }
332  else
333  { // We're still in the group :
334  m_hit = m_hit_group_buffer[m_group_read_cursor];
335  m_group_read_cursor++;
336  }
337  }
338  else
339  { // We aren't in a group
340  // Here is the "normal" data reading :
341  m_data = faster_file_reader_next(m_reader);
342  if (!m_data) return false; // This tags the end of the file
343  ReadData(m_data);
344  }
345  }
346  return m_data;
