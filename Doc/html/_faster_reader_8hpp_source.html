319 bool FasterReader::ReadGroup()
320 {
321  m_write = false;
322  while(!m_write)
323  {
324  if (m_inGroup)
325  { // We currently are reading a group
326  // if (m_group_read_cursor > m_group_write_cursor) // Might be losing the last hit with "==" condition ...
327  if (m_group_read_cursor == m_group_write_cursor)
328  { // We are getting out of the group, time to read the next data or group of data
329  m_inGroup = false;
330  // Reset m_hit_group_buffer cursors
331  m_group_read_cursor = 0;
332  m_group_write_cursor = 0;
333  m_data = faster_file_reader_next(m_reader);
334  if (!m_data) return false; // This tags the end of the data
335  ReadData(m_data);
336  }
337  else
338  { // We're still in the group :
339  m_hit = m_hit_group_buffer[m_group_read_cursor];
340  m_group_read_cursor++;
341  }
342  }
343  else
344  { // We aren't in a group
345  m_data = faster_file_reader_next(m_reader);
346  if (!m_data) return false; // This tags the end of the file
347  ReadData(m_data);
348  }
349  }

356 bool FasterReader::ReadData(faster_data_p const & _data)
357 {
358  m_hit->label = faster_data_label(_data);
359 #ifndef QDC1MAX
360  m_hit->nrj2 = 0; // In order to clean the data, as nrj2 never gets cleaned if there was QDC2 in the previous hit

363  if (m_alias == GROUP_TYPE_ALIAS)
364  {// Creates a group reader

367  ReadDataGroup(_data);

369  std::cout << "COMPILE IN GROUP MODE !!!" << std::endl;
370  #endif //FASTER_GROUP
371  }
372  m_hit -> time = faster_data_hr_clock_ns(_data) * 1000;// Stores the time in ps
373  m_write = switch_alias(m_alias, _data);
374 #ifdef FASTER_GROUP
375  if (m_inGroup && m_write)
376  {// We recursively end up here using ReadDataGroup() method (because m_inGroup is set to true)
377  // Here we fill in the m_hit_group_buffer with the hits extracted by the previous lines
378  // Hence, we convert a Data buffer (recursively readed) into a Hit buffer,
379  // that one can read hit by hit using the ReadGroup method
380  m_hit_group_buffer[m_group_write_cursor] = m_hit;

382  }

394 void FasterReader::ReadDataGroup(faster_data_p const & _data)
395 {
396  // Extracts all information in the "Group Data", which is the first "Data" of a group
397  unsigned short lsize = faster_data_load_size (_data);
398  void* group_buffer = faster_data_load_p (_data);
