219bool FasterReader::Initialize()
220{
221 #ifdef FASTER_GROUP
222 m_hit_group_buffer.resize(5000, &m_empty_hit); //If the number of hits in one group exceeds 5000 then it will crash
223 m_group_write_cursor = 0; m_group_read_cursor = 0;
224 #endif //FASTER_GROUP
225
226 // Check if the file can be open and read :
227 if (m_filename == "")
228 {
229 std::cout << "No file " << m_filename << std::endl;
230 return false;
231 }
232 std::ifstream file(m_filename);
233 if (!file.is_open())
234 {
235 std::cout << "File " << m_filename << " not found..." << std::endl;
236 return false;
237 }
238
239 else if (!file.good())
240 {
241 std::cout << "File " << m_filename << " not good..." << std::endl;
242 return false;
243 }
244 file.close();
245
246 // Check the extension of the file, then initialise the reader :
247 if (extension(m_filename) == "fast")
248 {
249 std::cout << "Reader set to " << m_filename << std::endl;
250 return InitializeReader();
251 }
252 else
253 {
254 std::cout << "File not a .fast file !!" << std::endl;
255 m_kReady = false; return false;
256 }
257 return false;
258}

298bool FasterReader::ReadSimple()
299{
300 // m_write variable is used to make sure only the handled aliases are read (see switch_alias definition)
301 m_write = false;
302 // faster_file_reader_next returns a value > 0 (->true in boolean) if the read was successfull (fasterac library)
303 // Ergo, the loop continues as long as the data is not one of the handled alias
304 while(!m_write && (m_data = faster_file_reader_next(m_reader)))
305 {
306 ReadData(m_data);

330bool FasterReader::ReadGroup()
331{
332 m_write = false;
333 while(!m_write)
334 {
335 if (m_inGroup)
336 { // We currently are reading a group
337 // if (m_group_read_cursor > m_group_write_cursor) // Might be losing the last hit with "==" condition ...
338 if (m_group_read_cursor == m_group_write_cursor)
339 { // We are getting out of the group, time to read the next data or group of data
340 m_inGroup = false;
341 // Reset m_hit_group_buffer cursors
342 m_group_read_cursor = 0;
343 m_group_write_cursor = 0;
344 m_data = faster_file_reader_next(m_reader);
345 if (!m_data) return false; // This tags the end of the data
346 ReadData(m_data);
347 }
348 else
349 { // We're still in the group :
350 m_hit = m_hit_group_buffer[m_group_read_cursor];
351 m_group_read_cursor++;
352 }
353 }
354 else
355 { // We aren't in a group
356 m_data = faster_file_reader_next(m_reader);
357 if (!m_data) return false; // This tags the end of the file
358 ReadData(m_data);
359 }
360 }
361 return m_data;
362}

367void FasterReader::ReadData(faster_data_p const & _data)
368{
369 m_hit->label = faster_data_label(_data);
370#ifdef QDC2
371 m_hit->nrj2 = 0; // In order to clean the data, as nrj2 never gets cleaned if there was QDC2 in the previous hit
372#endif //QDC2
373 m_alias = faster_data_type_alias(_data);
374 if (m_alias == GROUP_TYPE_ALIAS)
375 {// Creates a group reader
376 #ifdef FASTER_GROUP
377 // When read a label of a group, then go to ReadDataGroup() to fill in m_hit_group_buffer recursively
378 ReadDataGroup(_data);
379 #else
380 std::cout << "COMPILE IN GROUP MODE !!!" << std::endl;
381 #endif //FASTER_GROUP
382 }
383 m_hit -> time = faster_data_hr_clock_ns(_data) * 1000;// Stores the time in ps
384 m_write = switch_alias(m_alias, _data);
385 #ifdef FASTER_GROUP
386 if (m_inGroup && m_write)
387 {// We recursively end up here using ReadDataGroup() method (because m_inGroup is set to true)
388 // Here we fill in the m_hit_group_buffer with the hits extracted by the previous lines
389 // Hence, we convert a Data buffer (recursively readed) into a Hit buffer,
390 // that one can read hit by hit using the ReadGroup method
391 m_hit_group_buffer[m_group_write_cursor] = m_hit;
392 m_group_write_cursor++;
393 }
394 #endif //FASTER_GROUP
395}

404void FasterReader::ReadDataGroup(faster_data_p const & _data)
405{
406 // Extracts all information in the "Group Data", which is the first "Data" of a group
407 unsigned short lsize = faster_data_load_size (_data);
408 void* group_buffer = faster_data_load_p (_data);
409 auto group_reader = faster_buffer_reader_open(group_buffer, lsize);
410 faster_data_p group_data;
411 // Loop over data in data group buffer :
412 m_inGroup=true;
413 while( (group_data = faster_buffer_reader_next (group_reader)) )
414 {
415 // Recursive call to ReadData in order to read all the data stored in group_buffer and fill m_hit_group_buffer
416 ReadData(group_data);
417 }
418 faster_buffer_reader_close(group_reader);
419}
