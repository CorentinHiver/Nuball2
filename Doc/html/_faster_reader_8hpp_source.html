201bool FasterReader::Initialize()
202{
203 #ifdef FASTER_GROUP
204 m_hit_group_buffer.resize(5000, &m_empty_hit); //If the number of hits in one group exceeds 5000 then it will crash
205 m_group_write_cursor = 0; m_group_read_cursor = 0;
206 #endif //FASTER_GROUP
207
208 // Check if the file can be open and read :
209 if (m_filename == "")
210 {
211 std::cout << "No file " << m_filename << std::endl;
212 return false;
213 }
214 std::ifstream file(m_filename);
215 if (!file.is_open())
216 {
217 std::cout << "File " << m_filename << " not found..." << std::endl;
218 return false;
219 }
220
221 else if (!file.good())
222 {
223 std::cout << "File " << m_filename << " not good..." << std::endl;
224 return false;
225 }
226 file.close();
227
228 // Check the extension of the file, then initialise the reader :
229 if (extension(m_filename) == "fast")
230 {
231 std::cout << "Reader set to " << m_filename << std::endl;
232 return InitializeReader();
233 }
234 else
235 {
236 std::cout << "File not a .fast file !!" << std::endl;
237 m_kReady = false; return false;
238 }
239 return false;
240}

278bool FasterReader::ReadSimple()
279{
280 // m_write variable is used to make sure only the handled aliases are read (see switch_alias definition)
281 m_write = false;

310bool FasterReader::ReadGroup()
311{
312 m_write = false;
313 while(!m_write)
314 {
315 if (m_inGroup)
316 { // We currently are reading a group
317 // if (m_group_read_cursor > m_group_write_cursor) // Might be losing the last hit with "==" condition ...
318 if (m_group_read_cursor == m_group_write_cursor)
319 { // We are getting out of the group, time to read the next data or group of data
320 m_inGroup = false;
321 // Reset m_hit_group_buffer cursors
322 m_group_read_cursor = 0;
323 m_group_write_cursor = 0;
324 m_data = faster_file_reader_next(m_reader);
325 if (!m_data) return false; // This tags the end of the data
326 ReadData(m_data);
327 }
328 else
329 { // We're still in the group :
330 m_hit = m_hit_group_buffer[m_group_read_cursor];
331 m_group_read_cursor++;
332 }
333 }
334 else
335 { // We aren't in a group
336 m_data = faster_file_reader_next(m_reader);
337 if (!m_data) return false; // This tags the end of the file
338 ReadData(m_data);
339 }
340 }
341 return m_data;
342}

347bool FasterReader::ReadData(faster_data_p const & _data)
348{
349 m_hit->label = faster_data_label(_data);
350#ifndef QDC1MAX
351 m_hit->nrj2 = 0; // In order to clean the data, as nrj2 never gets cleaned if there was QDC2 in the previous hit
352#endif //QDC1MAX
353 m_alias = faster_data_type_alias(_data);
354 if (m_alias == GROUP_TYPE_ALIAS)
355 {// Creates a group reader
356 #ifdef FASTER_GROUP
357 // When read a label of a group, then go to ReadDataGroup() to fill in m_hit_group_buffer recursively
358 ReadDataGroup(_data);
359 #else
360 std::cout << "COMPILE IN GROUP MODE !!!" << std::endl;
361 #endif //FASTER_GROUP
362 }
363 m_hit -> time = faster_data_hr_clock_ns(_data) * 1000;// Stores the time in ps
364 std::cout << std::setprecision(20) << faster_data_hr_clock_ns(_data) << std::endl;
365 m_write = switch_alias(m_alias, _data);
366#ifdef FASTER_GROUP
367 if (m_inGroup && m_write)
368 {// We recursively end up here using ReadDataGroup() method (because m_inGroup is set to true)
369 // Here we fill in the m_hit_group_buffer with the hits extracted by the previous lines
370 // Hence, we convert a Data buffer (recursively readed) into a Hit buffer,
371 // that one can read hit by hit using the ReadGroup method
372 m_hit_group_buffer[m_group_write_cursor] = m_hit;
373 m_group_write_cursor++;
374 }
375#endif //FASTER_GROUP
376 return m_write;
377}

386void FasterReader::ReadDataGroup(faster_data_p const & _data)
387{
388 // Extracts all information in the "Group Data", which is the first "Data" of a group
389 unsigned short lsize = faster_data_load_size (_data);
390 void* group_buffer = faster_data_load_p (_data);
391 auto group_reader = faster_buffer_reader_open(group_buffer, lsize);
392 faster_data_p group_data;
393 // Loop over data in data group buffer :
394 m_inGroup=true;
395 while( (group_data = faster_buffer_reader_next (group_reader)) )
396 {
397 // Recursive call to ReadData in order to read all the data stored in group_buffer and fill m_hit_group_buffer
398 ReadData(group_data);
399 }
400 faster_buffer_reader_close(group_reader);
401}
