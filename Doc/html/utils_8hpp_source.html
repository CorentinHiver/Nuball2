589  tree -> GetEntry(i);
590  // if (i<20) std::cout << label << "\t" << TimeStamp << std::endl;
591  TimeStampBuffer[i]=TimeStamp;

595  NewIndex[0]=0;
596  for (j=0; j<NHits;j++)

599  a=TimeStampBuffer[j]; //Focus on this time stamp
600  i=j;

604  NewIndex[i]=NewIndex[i-1];

606  }

826  }
827  }

829  CL [n_cond][1]-=1;
830  n_cond++;
