38Bool_t CoincBuilder2::build(Hit const & hit)
39{//return true when a coincidence is ready to be processed
40 if(m_event->mult>255) reset();
41 if (!coincON)
42 {// If no coincidence has been detected in previous iteration :
43 if (coincidence(hit))
44 {// Case 1 :
45 // The previous and current hit are in the same event.
46 // In next call, we'll check if the next hits are in the event or not (cases 3 or 4)
47 m_event -> clear();
48 m_event -> push_back(m_last_hit);
49 m_event -> push_back(hit);
50 coincON = true; // Open the event
51 m_status = 1; // Now, the event is being filled
52 }
53 else
54 {// Case 2 :
55 // The last and current hits aren't in the same event.
56 // This means either the last hit is filled with an empty hit if last hit closed an event,
57 // or a single hit : a lonely hit alone in the time window around its RF.
58 m_single_hit = m_last_hit; // If last hit is not in time window of last hit, then
59 m_last_hit = hit; // Building next event based on the last hit
60 m_status = 0; // The current event is empty !
61 }
62 }
63 else
64 {// ... there ! Coincidence already detected, check if current hit is out of currently building event
65 if (coincidence(hit))
66 {// Hit in coincidence with the previous hits
67 m_event -> push_back(hit);
68 }
69 else
70 {// Hit out of coincidence. Next call, go back to first condition.
71 m_last_hit = hit; // Build next event based on the current hit, that is not in the event.
72 coincON = false;
73 m_status = 2; // The current event is full
74 return true; // Now the event is complete and can be treated
75 }
76 }
77 return false;
