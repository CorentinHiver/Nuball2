121 file.close();
122 if (extension(m_filename) == "root")
123 {
124 m_rootORfast = "root";
125 m_quick_rootOrFast = 'r';
126 std::cout << "Reader set to " << m_filename << std::endl;
127 return InitializeRootRead();

131 #ifndef NO_FASTERAC

139 }
140 else

142 std::cout << "File not a .root nor a .fast file" << std::endl;
