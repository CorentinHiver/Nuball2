171 template <typename T>
172 std::vector<int> & bubble_sort(std::vector<T> const & vector, std::vector<int> & ordered_indexes)
173 {
174  // Verifications :
175  if (vector.size() == 0) {printC(RED, "In bubble_sort(vector, ordered_indexes) : vector size is zero !", RESET); return ordered_indexes;}
176  if (vector.size() != ordered_indexes.size()) ordered_indexes.resize(vector.size());
177 
178  // Initialisations :
179  T v = vector[0];
180  ordered_indexes[0] = 0;
181  size_t j = 0;
182 
183  // Loop through the vector :
184  for (size_t i = 0;i<vector.size(); i++)
185  {
186  // Initial guess : the ith ordered_indexes's index corresponds to the vector's ith index
187  // (e.g. the 5th bin has initial value 5 (ordered_indexes[5] = 5))
188  ordered_indexes[i] = i;
189 
190  // Check this assumption : v holds the value of ith value of vector
191  v = vector[i];
192 
193  // Find the true jth index of the ith vector's index
194  j = i;
195 
196  // Loop goes on until the (j-1)th vector's value is lower than the ith value
197  while((j>0) && vector[ordered_indexes[j-1]] > v)
198  {
199  // If the (j-1)th value is higher than the ith value then switch the indexes.
200  ordered_indexes[j] = ordered_indexes[j-1];
201  --j;
202  }
203 
204  // Save the correct position of the index
205  ordered_indexes[j] = i;
206  }
207 
208  // Note that this method is iterative : if the 1st value is higher than the 2nd,
209  // ordered_index[0] = 1 and ordered_index[1] = 0. If now the 3rd value is higher than
210  // the 2nd but lower than the 1st (for i = 2), vector[ordered_index[1]] > v
211  // but vector[ordered_index[0]] < v : the result is indeed {1, 2, 0}
212 
213  return ordered_indexes;
214 }
