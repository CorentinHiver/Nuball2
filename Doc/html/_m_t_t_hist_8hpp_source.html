227{
228 if (!m_exists || !m_collection.size() || m_collection[0] -> IsZombie() || m_collection[0] -> Integral() < 1)
229 {
230 m_exists = false;
231 delete m_merged;

244 m_merged = static_cast<THist*> (m_collection[0]->Clone(m_str_name.c_str()));
245 for (size_t i = 1; i<m_collection.size(); i++) m_merged -> Add(m_collection[i]);
246 if (m_file) m_file -> cd(); //To return to the scope of any potential TFile
