255void MTTHist<THist>::Merge()
256{
257 if (!m_exists || !m_collection.size() || m_collection[0] -> IsZombie() || m_collection[0] -> Integral() < 1)
258 {
259 m_exists = false;
260 delete m_merged;
261 m_is_merged = false;
262 }
263 else

267 m_merged = m_collection[0];

269 else
270 {

272 if (m_file) gROOT -> cd(); //To get out of scope of any potential TFile
273 m_merged = static_cast<THist*> (m_collection[0]->Clone(m_str_name.c_str()));
274 for (size_t i = 1; i<m_collection.size(); i++) m_merged -> Add(m_collection[i]);
275 if (m_file) m_file -> cd(); //To return to the scope of any potential TFile
