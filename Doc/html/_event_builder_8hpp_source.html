44Bool_t EventBuilder::build(Hit const & hit)
45{
46 if(m_event->mult>255) reset(); // 255 is the maximum number of hits allowed inside of an event
47 if (!coincON)
48 { // If no coincidence has been detected in previous iteration :
49 if ( (hit.time-RF_ref_time) < static_cast<ULong64_t>(m_rf->period) )
50 {// Case 1 :
51 // The previous and current hit are in the same event.
52 // In next call, we'll check if the next hits are in the event or not (cases 3 or 4)
53 m_event -> clear();
54 #ifdef PREPROMPT
55 if ( (RF_ref_time-m_single_hit.time) < static_cast<ULong64_t>(m_rf->period))
56 m_event -> push_back(m_single_hit);
57 #endif //PREPROMPT
58 m_event -> push_back(m_last_hit);
59 m_event -> push_back(hit);
60 m_last_hit = m_empty_hit; // m_last_hit is cleared
61 coincON = true; // Open the event
62 m_status = 1; // The event can be filled with potential additionnal hits
63 }
64 else
65 {// Case 2 :
66 // The last and current hits aren't in the same event.
67 // This means either the last hit is filled with an empty hit if last hit closed an event,
68 // or a single hit : a lonely hit, alone in the time window around its RF.
69 m_single_hit = m_last_hit;
70 // The current hit is set to be the last hit for next call :
71 this -> set_last_hit(hit);
72 m_status = 0; // No event detected
73 }
74 }
75
76 else
77 { // If the two previous hits are in the same event, checking if the current hit is :
78 if ( (hit.time-RF_ref_time) < static_cast<ULong64_t>(m_rf->period) )
79 {// Case 3 :
80 // print(hit.time, hit.label, (hit.time-RF_ref_time)/1000., m_rf->period/1000.);
81 // The current hit belongs to the event
82 m_event -> push_back(hit);
83 // NB: m_status still equals to 1
84 }
85 else
86 {// Case 4 :
87 // The current hit is outside of the event : this hit closes the event :
88 coincON = false; // The coincidence is closed
89 // Therefore, m_event is full with all the hits of the event :
90 m_status = 2; // The event is complete
91 // The current hit is set to be the last hit for next call :
92 m_last_hit = hit;
93 return true; // The event is ready to be processed
94 }
95 }
96 return false; // No processing of the data
97}
