139Bool_t Evt_build::build(Hit const & hit)
140{//return true when an event is ready to be processed
141 if (m_status == 2 || mult>254) this->flush(); //because nb_evts is on uchar so < 255
142 if (!coincON)
143 {// No coincidence going on. If there is, go ...
144 if (coincidence(hit))
145 {// Bingo, coincidence detected ! Next call, go ...
146 m_buffer -> push_back( m_last_hit );
147 m_buffer -> push_back( hit );
148 mult = 2;
149 coincON = true;
150 m_status = 1;
151 }
152 else
153 {// No coincidence detected...
154 m_single_hit = m_last_hit;
155 m_last_hit = hit; //store the hit for next loop
156 mult = 1;
157 n_evt++;
158 m_status = 0;
159 }
160 }
161 else
162 {// ... there ! Coincidence already detected, check if it ends
163 if (coincidence(hit))
164 {// Hit in coincidence with the previous hits
165 mult++;
166 m_buffer -> push_back(hit);
167 }
168 else
169 {// Hit out of coincidence. Next call, go back to first condition.
170 n_evt++;
171 coincON = false;
172 m_last_hit = hit;
173 m_status = 2;
174 return true;
175 }
176 }
177 return false;
178}
