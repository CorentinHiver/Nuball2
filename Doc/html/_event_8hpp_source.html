170void Event::writeTo(TTree * tree, std::string const & options)
171{
172 if (!tree) {print("Output tree at address 0x00 !"); return;}
173 for (auto const & e : options)
174 {
175 switch (e)
176 {
177 case ('l') : write.l = true; break;
178 case ('t') : write.t = true; break;
179 case ('T') : write.T = true; break;
180 case ('n') : write.E = true; break;
181 case ('N') : write.E2 = true; break;
182 case ('p') : write.p = true; break;
183 case ('R') : write.RFt = true; break;
184 case ('P') : write.RFp = true; break;
185 }
186 }
187
188 tree -> ResetBranchAddresses();
189
190 tree -> Branch("mult" , &mult);
191 if ( write.l ) tree -> Branch("label" , &labels , "label[mult]/s" );
192 if ( write.t ) tree -> Branch("time" , &times , "time[mult]/l" );
193 if ( write.T ) tree -> Branch("Time" , &time2s , "Time[mult]/F");
194 if ( write.E ) tree -> Branch("nrj" , &nrjs , "nrj[mult]/F" );
195 if ( write.E2 ) tree -> Branch("nrj2" , &nrj2s , "nrj2[mult]/F" );
196 if ( write.p ) tree -> Branch("pileup" , &pileups, "pileup[mult]/O");
197 if ( write.RFt ) tree -> Branch("RFtime" , &RFtime );
198 if ( write.RFp ) tree -> Branch("RFperiod" , &RFperiod);
199
200 tree -> SetBranchStatus("*",true);
