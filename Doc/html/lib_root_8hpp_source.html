50void alignator(TTree * tree, int *NewIndex)
51{
52 int const NHits = tree -> GetEntries();
53 tree -> SetBranchStatus("*", false);// Disables all the branches readability
54 tree -> SetBranchStatus("time", true);// Read only the time
55 // tree -> SetBranchStatus("label", true);// Read only the time
56 std::vector<ULong64_t> TimeStampBuffer(NHits,0);
57 ULong64_t TimeStamp = 0; tree->SetBranchAddress("time", &TimeStamp);
58 // UShort_t label = 0; tree->SetBranchAddress("label", &label);
59 for (int i = 0; i<NHits; i++)
60 {
61 tree -> GetEntry(i);
62 // if (i<20) std::cout << label << "\t" << TimeStamp << std::endl;
63 TimeStampBuffer[i]=TimeStamp;
64 }
65 int i = 0, j = 0;
66 ULong64_t a = 0;
67 NewIndex[0]=0;
68 for (j=0; j<NHits;j++)
69 {
70 NewIndex[j]=j;
71 a=TimeStampBuffer[j]; //Focus on this time stamp
72 i=j;
73 // Find the place to insert it amongst the previously sorted
74 while((i > 0) && (TimeStampBuffer[NewIndex[i-1]] > a))
75 {
76 NewIndex[i]=NewIndex[i-1];
77 i--;
78 }
79 NewIndex[i]=j;
80 }
81 tree -> SetBranchStatus("*", true); //enables again the whole tree
82}
