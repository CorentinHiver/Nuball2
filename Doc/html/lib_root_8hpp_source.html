176 {
177  auto const NHits = tree -> GetEntries();
178  tree -> SetBranchStatus("*", false);// Disables all the branches readability
179  tree -> SetBranchStatus("time", true);// Read only the time
180 
181  std::vector<ULong64_t> TimeStampBuffer(NHits,0);
182  ULong64_t TimeStamp = 0; tree->SetBranchAddress("time", &TimeStamp);
183 
184  // First creates a buffer of all the timestamps :
185  for (int i = 0; i<NHits; i++)
186  {
187  tree -> GetEntry(i);
188  TimeStampBuffer[i]=TimeStamp;
189  }
190 
191  // Then computes the correct order :
192  int i = 0, j = 0;
193  ULong64_t a = 0;
194  NewIndex[0]=0;
195  for (j=0; j<NHits;j++)
196  {
197  NewIndex[j]=j;
198  a=TimeStampBuffer[j]; //Focus on this time stamp
199  i=j;
200  // Find the place to insert it amongst the previously sorted timestamps
201  while((i > 0) && (TimeStampBuffer[NewIndex[i-1]] > a))
202  {
203  NewIndex[i]=NewIndex[i-1];
204  i--;
205  }
206  NewIndex[i]=j;
207  }
208  tree -> SetBranchStatus("*", true); //enables again the whole tree to be read
209 }
